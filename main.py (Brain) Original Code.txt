from flask import Flask, request
import logging

# 1. This creates the "Brain" app
app = Flask(__name__)

# 2. This sets up your "Receipt Book" (logs.txt)
logging.basicConfig(
    filename='logs.txt',
    level=logging.INFO,
    format='%(asctime)s - %(message)s'
)

# 3. This is the "Front Door" - Visit this in your browser to check if it's awake
@app.route('/')
def home():
    return "The Brain is AWAKE and listening!"

# 4. This is the "Scout Entrance" - This is where your VS Code data enters
@app.route('/scout', methods=['POST'])
def receive_data():
    # Capture the data sent from VS Code
    data = request.json
    door_name = data.get("door", "Unknown Door")
    status = data.get("status", "Unknown Status")

    # Create the alert message
    alert = f"EVENT: {door_name} changed to {status}"
    
    # Print to the Replit console and save to logs.txt
    print(alert)
    logging.info(alert)

    return {"message": "Data received by Brain!"}, 200

# 5. The "Power Switch" to turn it on
if __name__ == "__main__":
    # Replit needs host 0.0.0.0 and port 8080 to talk to the outside world
    app.run(host='0.0.0.0', port=8080)
